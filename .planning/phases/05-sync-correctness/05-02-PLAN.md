2. **src/core/status.ts** -- Add a checkSyncHealth function:

```typescript
function checkSyncHealth(db: Database.Database): SyncHealth | null {
  try {
    // Pending count
    const pending = db.prepare("SELECT COUNT(*) as cnt FROM sync_failures WHERE status = 'pending'").get();
    // Parked count
    const parked = db.prepare("SELECT COUNT(*) as cnt FROM sync_failures WHERE status = 'parked'").get();
    // Oldest pending failure
    const oldest = db.prepare("SELECT MIN(first_failed_at) as oldest FROM sync_failures WHERE status = 'pending'").get();
    // Total retry attempts
    const retries = db.prepare("SELECT COALESCE(SUM(attempt_count), 0) as total FROM sync_failures").get();
    // Last successful sync (store this in cortex_meta as 'last_successful_sync_at')
    const lastSync = db.prepare("SELECT value FROM cortex_meta WHERE key = 'last_successful_sync_at'").get();

    // Format oldest age as human-readable relative time
    // Calculate difference between now and oldest.oldest, format as "Xh Ym" or "Xd Yh"
    return { pending_count, parked_count, oldest_pending_age, last_successful_sync, total_retry_attempts };
  } catch {
    return null; // table doesn't exist yet
  }
}
```
