```typescript
export interface ScoringExplanation {
  observation_id: number;
  fts_score: number | null;       // Raw FTS rank (from ftsRank)
  vector_score: number | null;    // Raw vector rank (from vecRank)
  rrf_combined: number;           // Score after RRF fusion (the initial r.score from hybridSearch)
  graph_expansion: number;        // 0 for direct results; for graph-expanded: the discount factor applied (0.5x lowest)
  lifecycle_confidence: number;   // Output of computeConfidence() -- the multiplier applied
  recency_boost: number;          // The 1 + 0.5*exp(-ageDays/30) factor
  final_composite: number;        // The final score after all multipliers
  source: string;                 // "both" | "fts" | "vec" | "fts-fallback" | "graph-expanded"
}

export interface ExcludedCandidate {
  observation_id: number;
  title: string;
  reason: string;    // "deprecated" | "superseded" | "lifecycle_zero_confidence"
  original_score: number;
}

export interface SearchExplainResult {
  results: Array<SearchResult & { explain?: ScoringExplanation }>;
  excluded: ExcludedCandidate[];
  metadata: {
    total_candidates: number;
    total_after_lifecycle_filter: number;
    total_after_graph_expansion: number;
    search_source: "hybrid" | "fts-fallback";
  };
}
```

**In src/core/search.ts**, modify the search pipeline:

1. Add `explain?: boolean` to the `SearchOptions` interface.

2. Modify `filterLifecycle()` to optionally return excluded candidates:
   - Add a second return value or change to return `{ kept: SearchResult[], excluded: ExcludedCandidate[] }` when explain is true.
   - For each excluded result, record: observation_id, title, reason (look up obs_lifecycle status), original_score.

3. Modify `enrichWithLifecycle()` to optionally attach per-result explanation:
   - When explain is true, compute and store intermediate values BEFORE multiplying into composite:
     - `rrf_combined` = the input `r.score` (before lifecycle/recency)
     - `lifecycle_confidence` = output of `computeConfidence()`
     - `recency_boost` = the `1 + 0.5 * exp(-ageDays/30)` value
     - `final_composite` = `rrf_combined * lifecycle_confidence * recency_boost`
   - Attach as `.explain` property on each result.

4. Modify `graphExpandResults()` to set `graph_expansion` on explain data:
   - For direct results: `graph_expansion = 0`
   - For graph-expanded results: `graph_expansion = graphScore` (the 0.5x lowest value)

5. Modify the main `search()` function:
   - Thread `explain` option through the pipeline.
   - When explain is true, collect excluded candidates from filterLifecycle.
   - Build `SearchExplainResult` metadata (candidate counts at each stage, search_source).
   - Return explain data as a second text block in the MCP content array (JSON formatted).

6. Format explain output: When explain is true, append a second content block:
   ```
   { type: "text", text: JSON.stringify(explainResult, null, 2) }
   ```
   The first content block remains the normal human-readable formatted results.

IMPORTANT: When explain is false/undefined (the default), behavior is IDENTICAL to current -- no performance impact, no extra queries. The explain path only activates when explicitly requested.
